# 常用命令

# 辅助命令

docker version
docker info
docker --help

# images镜像命令

# 查看本机中所有镜像
docker images [options] [name]
	-a：列出所有镜像（包含中间映像层）
	-q：只显示镜像id
	name：只搜索相关的iamges，相当于过滤条件
	
docker images
docker images -aq
docker images -aq mysql
docker images mysql
# 搜索镜像
docker search [options] 镜像名

docker search mysql
# 下载镜像
docker pull 镜像名[:TAG|@DIGEST]

docker pull mysql:5.7.34
# 删除镜像

必须先停止相关容器，强制删除会一起将容器删除

docker image rm 镜像名:tag | 镜像id
docker rmi 镜像名:tag | 镜像id
	-f：强制删除
	
docker image rm d1165f221234
docker image rm -f d11
docker rmi -f d11
docker rmi -f $(docker images -q)	# 使用组合命令删除所有镜像

# container容器命令

# 导入tar到本地仓库
docker load -i 名称.tar

docker load -i mysql-5.7.tar
# 查看正在运行的容器
docker ps [options]
	-a：正在运行的和历史运行过的容器
	-q：只显示容器id（正在运行的）
	
docker ps
docker ps -aq
# 运行容器
docker run [options] 镜像名:tag | 镜像id
	--name：为容器起一个别名
	-d：启动守护式容器（后台启动）
	-p：宿主机端口号:原始端口号 指定端口号启动，可以书写多个
	
docker run -d --name MyTomcat -p 8888:8080 tomcat:$tag	# 返回容器的运行id
# 停止|关闭|重启容器
docker start 容器名或id
docker restart 容器名或id
docker stop 容器名或id		# 容器瑞出前会做一些安全或保护操作
docker kill 容器名或id     	# 直接杀死，不管容器同不同意
# 删除容器
docker rm [options] 容器名或id
	-f：强制删除正在运行的容器
	
docker rm -f tomcat01
docker rm -f $(docker ps -aq)	# 组合删除所有容器
# 查看日志
docker logs [options] 容器名或id
	-f：实时展示，必须在容器启动后
	-tf：实时展示，加入时间戳
	--tail：只显示容器日志尾部的5行
	
docker logs -tf MyTomcat
# 查看容器内的进程
docker top 容器名或id
# 查看容器内部细节
docker inspect 容器名或id
# 进入容器
docker exec [options] 容器名或id bash
	-i：以交互的模式运行容器
	-t：分配一个伪终端
	
docker exec -it 6b4 bash
# 容器与主机之间复制文件
docker cp 文件|目录 容器名或id：容器内资源路径
docker cp 容器名或id：容器内资源路径 主机目录

docker cp a.txt 6b4:/usr/local/tomcat
docker cp 6b4:/usr/local/tomcat/RUNNING.txt ./RUNNING.txt	# 复制到当前路径，名为RUNNING.txt
# 数据卷实现共享目录

类似于VMware中虚拟机与物理主机的共享文件夹

docker run -v 宿主机的路径|任意别名:/容器内的路径 镜像名
docker run -v 宿主机的路径|任意别名:/容器内的路径:ro 镜像名	# ro容器只能读不能写

docker run -d -p 8080:8080 --name tomcat01 -v /root/apps:/usr/local/tomcat/webapps tomcat:8.0-jre8.0		# 只能在启动时指定数据卷且启动后会覆盖容器内目录下原有文件
docker run -d -p 8080:8080 --name tomcat01 -v aa:/usr/local/tomcat/webapps tomcat:8.0-jre8.0		# aa代表一个数据卷名字，随便写，不存在则自动创建数据卷同时自动映射宿主机中某个目录，同时将容器目录文件复制到宿主机中

好处：配置映射后，即使将容器删除了，但写的内容还会保存在宿主机中

# 容器打包成镜像

容器打包期间会暂停运行

docker commit -m "描述信息" -a "作者信息" （容器id或者名称） 打包的镜像名称:标签

docker commit -m "test.war" -a "yt" 91f test-tomcat:1.0
# 将镜像打成tar包
docker save 镜像名 -o  名称.tar

docker save test-tomcat:1.0 -o test-tomcat-1.0.tar
# 载入镜像
docker load -i 名称.tar

docker load -i test-tomcat-1.0.tar
