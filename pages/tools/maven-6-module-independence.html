# 创建

3个Maven项目，分别为Dao、Service、Common，一个SpringBoot项目Web（含controller）

# 依赖关系

  1. common
  2. dao --> common
  3. service --> dao, common
  4. web --> service, common

# POM

common

<?xml version="1.0" encoding="UTF-8"?>
<project xmlns="http://maven.apache.org/POM/4.0.0"
         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">

    <modelVersion>4.0.0</modelVersion>

    <groupId>com.yangtao</groupId>
    <artifactId>Maven-SpringBoot-MultiModule-Common</artifactId>
    <version>1.0.0</version>

    <properties>
        <maven.compiler.source>8</maven.compiler.source>
        <maven.compiler.target>8</maven.compiler.target>
    </properties>

</project>

web

<?xml version="1.0" encoding="UTF-8"?>
<project xmlns="http://maven.apache.org/POM/4.0.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
         xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 https://maven.apache.org/xsd/maven-4.0.0.xsd">
    <modelVersion>4.0.0</modelVersion>
    <parent>
        <groupId>org.springframework.boot</groupId>
        <artifactId>spring-boot-starter-parent</artifactId>
        <version>2.5.0</version>
        <relativePath/> <!-- lookup parent from repository -->
    </parent>
    <groupId>com.yangtao</groupId>
    <artifactId>Maven-SpringBoot-MultiModule-Web</artifactId>
    <version>0.0.1-SNAPSHOT</version>
    <name>Maven-SpringBoot-MultiModule-Web</name>
    <description>Maven-SpringBoot-MultiModule-Web</description>

    <!-- 版本集中管理 -->
    <properties>
        <java.version>1.8</java.version>
        <spring-boot-starter.version>2.5.0</spring-boot-starter.version>
        <spring-boot-starter-test.version>2.5.0</spring-boot-starter-test.version>
        <spring-boot-starter-web.version>2.5.0</spring-boot-starter-web.version>
        <lombok.version>1.18.20</lombok.version>
    </properties>
    <dependencies>
        <dependency>
            <groupId>com.yangtao</groupId>
            <artifactId>Maven-SpringBoot-MultiModule-Common</artifactId>
            <version>1.0.0</version>
        </dependency>
        <dependency>
            <groupId>com.yangtao</groupId>
            <artifactId>Maven-SpringBoot-MultiModule-Service</artifactId>
            <version>1.0.0</version>
        </dependency>


        <dependency>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-starter</artifactId>
            <version>${spring-boot-starter.version}</version>
        </dependency>

        <dependency>
            <groupId>org.projectlombok</groupId>
            <artifactId>lombok</artifactId>
            <optional>true</optional>
            <version>${lombok.version}</version>
        </dependency>
        <dependency>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-starter-test</artifactId>
            <scope>test</scope>
            <version>${spring-boot-starter-test.version}</version>
        </dependency>
        <dependency>
            <groupId>org.springframework.boot</groupId>
            <artifactId>spring-boot-starter-web</artifactId>
            <version>${spring-boot-starter-web.version}</version>
        </dependency>
    </dependencies>

    <build>
        <plugins>
            <plugin>
                <groupId>org.springframework.boot</groupId>
                <artifactId>spring-boot-maven-plugin</artifactId>
                <configuration> <!-- 指定该Main Class为全局的唯一入口 -->
                    <mainClass>
                        com.yangtao.MavenSpringBootMultiModuleWebApplication
                    </mainClass>
                    <layout>ZIP</layout>
                </configuration>
                <executions>
                    <execution>
                        <goals>
                            <goal>repackage</goal><!--可以把依赖的包都打包到生成的Jar包中-->
                        </goals>
                    </execution>
                </executions>
            </plugin>
        </plugins>
    </build>

</project>

这4个项目没有共同的父工程，入口是web，每次打包需要按顺序将项目install到本地仓库中，下一个依赖于该项目的项目才能install或package，最终web项目可以聚合所有的依赖。
