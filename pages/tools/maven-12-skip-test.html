# 手动配置

配置某个插件跳过全部测试

<plugins>
    <plugin>
        <artifactId>maven-surefire-plugin</artifactId>
        <version>2.22.1</version>
        <configuration>
            <skipTests>true</skipTests>
        </configuration>
    </plugin>
</plugins>

配置跳过部分插件

<plugins>
    <plugin>
        <artifactId>maven-surefire-plugin</artifactId>
        <version>2.22.1</version>
        <configuration>
            <includes>
                <include></include>
            </includes>
            <excludes>
                <exclude>
                    **/MavenSpringBootMultiModuleWebApplicationTests.java
                </exclude>
            </excludes>
        </configuration>
    </plugin>
</plugins>

# 命令

mvn 指令 -D skipTests
