# 获取加速地址

​ 访问阿里云登录自己账号查看docker镜像加速服务，地址：https://cr.console.aliyun.com/cn-shenzhen/instances/mirrors

# CentOS中配置
sudo mkdir -p /etc/docker
sudo tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://fg5ffem1.mirror.aliyuncs.com"]
}
EOF
sudo systemctl daemon-reload
sudo systemctl restart docker
# 验证镜像加速是否生效
docker info
