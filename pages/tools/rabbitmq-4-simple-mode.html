# 生产者

连接工厂 -> 设置IP用户名密码 -> 获取连接 -> 创建信道 -> 发送消息

public class Producer {
    //队列名称
    public static final String QUEUE_NAME = "hello";
    public static void main(String[] args) throws IOException, TimeoutException {
        //连接工厂
        ConnectionFactory factory = new ConnectionFactory();
        //设置连接IP
        factory.setHost("192.168.8.130");
        //设置用户名和密码
        factory.setUsername("admin");
        factory.setPassword("123");
        //创建连接
        Connection connection = factory.newConnection();
        //获取信道
        Channel channel = connection.createChannel();
        //生成队列：队列名  是否持久化  是否只有一个消费者消费  最后一个消费者端断开后是否自动删除
        channel.queueDeclare(QUEUE_NAME, false, false, false, null);
        //发消息
        String message = "hello world";
        //发送到哪个交换机  路由key（队列名）  其他参数信息  发送的消息（二进制）
        channel.basicPublish("", QUEUE_NAME, null, message.getBytes());
        System.out.println("消息发送成功");
    }
}

# 消费者

连接工厂 -> 设置IP用户名密码 -> 获取连接 -> 创建信道 -> 获取消息(成功和失败的回调)

public class Consumer {
    public static final String QUEUE_NAME = "hello";
    public static void main(String[] args) throws IOException, TimeoutException {
        //连接工厂
        ConnectionFactory factory = new ConnectionFactory();
        factory.setHost("192.168.8.130");
        factory.setUsername("admin");
        factory.setPassword("123");
        Connection connection = factory.newConnection();
        Channel channel = connection.createChannel();

        DeliverCallback deliverCallback = (consumerTag, message) -> {
            System.out.println("接收到的消息为：" + new String(message.getBody()));
        };

        CancelCallback cancelCallback = consumerTag -> {
            System.out.println("接收消息失败");
        };

        //消费哪个队列  消费成功后是否自动应答  消费者成功消费的回调  消费未成功消费的回调
        channel.basicConsume(QUEUE_NAME, true, deliverCallback, cancelCallback);
    }
}
