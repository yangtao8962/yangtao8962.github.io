# 概念

消息保存在磁盘上时出错，导致消息丢失，因此加入发布确认，只有发布确认了（消息已经保存在磁盘上了），MQ才会告知生产者消息已经持久化了。

  • 单个确认发布：同步，发一条，确认后再发下一条，会导致发布速度特别慢
  • 批量确认发布：同步，对所有消息一起确认，但是如果出现问题，会不知道哪条消息出了问题
  • 异步确认发布：最好的方式，利用回调函数来保证消息是否投递成功，比较复杂，一但出现消息丢失，可以马上知道哪条消息丢失了；两个线程：发布线程，监听器线程

# 代码测试

public class ConfirmMessage {
    public static void main(String[] args) throws Exception {
        signalPublish();
        batchPublish();
        asyncPublic();
    }

    //单个确认
    public static void signalPublish() throws Exception {
        Channel channel = RabbitMQUtils.getChannel();
        channel.confirmSelect();
        String queueName = UUID.randomUUID().toString();
        channel.queueDeclare(queueName, false, false, false, null);
        long start = System.currentTimeMillis();
        for (int i = 0; i < 1000; i++) {
            String message = i + "";
            channel.basicPublish("", queueName, null, message.getBytes());
            //发布确认，是否确认成功
            boolean b = channel.waitForConfirms();
        }
        long end = System.currentTimeMillis();
        System.out.println("单个确认耗时：" + (end - start) + " ms");
    }

    //批量确认
    public static void batchPublish() throws Exception {
        Channel channel = RabbitMQUtils.getChannel();
        channel.confirmSelect();
        String queueName = UUID.randomUUID().toString();
        channel.queueDeclare(queueName, false, false, false, null);
        long start = System.currentTimeMillis();
        //批量确认：每100条确认一次
        int batchSize = 1;
        for (int i = 0; i < 1000; i++) {
            String message = i + "";
            channel.basicPublish("", queueName, null, message.getBytes());
            //发布确认，是否确认成功
            if (batchSize % 100 == 0) {
                channel.waitForConfirms();
            }
            batchSize++;
        }
        long end = System.currentTimeMillis();
        System.out.println("批量确认耗时：" + (end - start) + " ms");
    }

    //异步确认
    public static void asyncPublic() throws Exception {
        Channel channel = RabbitMQUtils.getChannel();
        channel.confirmSelect();
        //线程安全有序的一个哈希表，适用于高并发的情况下：将序号与消息进行关联  删除已确认的消息
        ConcurrentSkipListMap<Long, String> outstandingConfirms = new ConcurrentSkipListMap<>();
        String queueName = UUID.randomUUID().toString();
        channel.queueDeclare(queueName, false, false, false, null);
        long start = System.currentTimeMillis();
        //消息确认成功的回调函数
        ConfirmCallback ackCallback = (deliveryTag, multiple) -> {
            //是否是批量
            if (multiple) {
                //已确认的消息进行清理
                ConcurrentNavigableMap<Long, String> confirmed = outstandingConfirms.headMap(deliveryTag);
                confirmed.clear();
            } else {
                outstandingConfirms.remove(deliveryTag);
            }
        };
        //消息确认失败的回调函数：消息的标识 是否为批量确认
        ConfirmCallback nackCallback = (deliveryTag, multiple) -> {
            String unconfirmedMessage = outstandingConfirms.get(deliveryTag);
            System.out.println("未确认的消息：" + unconfirmedMessage);
        };
        //监听器：监听消息的成功与失败
        channel.addConfirmListener(ackCallback, nackCallback);
        //批量发送消息
        for (int i = 0; i < 1000; i++) {
            String message = i + "";
            channel.basicPublish("", queueName, null, message.getBytes());
            //记录所有发送的消息
            outstandingConfirms.put(channel.getNextPublishSeqNo(), message);
        }
        long end = System.currentTimeMillis();
        System.out.println("异步确认耗时：" + (end - start) + " ms");
    }

}
