# where和having

两者都可以实现过滤记录的功能

where在聚合（group by）前筛选记录，having在聚合后筛选记录

如：查找更新时间在2021年后更新了且人口数 > 一千万的记录

select region,sum(population),update_at from tb where update_at>'2021-01-01 00:00:00' group by region having sum(population)>10000000;

where条件使用的是表中已存在的字段，而having条件总是使用聚合函数

# case when then end

两种用法

select 
	name,
	age,
	(case sex
		when '1' then '男'
		when '2' then '女'
		else '其他'
    end) 'sex'
from 
	user_info;

select 
	name,
	age,
	(case
		when sex = '1' then '男'
		when sex = '2' then '女'
		else '其他'
    end) 'sex'
from 
	user_info;

# concat(str1, str2, ...)

返回连接参数后的字符串，如果包含null，则整个结果为null

select concat('My','S','QL');
-> MySQL

select concat('My','S','QL',null);
-> null

# concat_ws(separator, str1, str2, ...)

第一个是分隔符，为null则整体结果为null，后续参数为null则直接忽略该参数

select concat_ws('-','My','S','QL');
-> My-S-QL

select concat_ws('-','My',null,'S','QL');
-> My-S-QL

select concat_ws(null,'My','S','QL');
-> null

# group_concat(col)

group_concat( [distinct] 查询字段 [order by 排序字段 asc/desc] [separator '分隔符'] );

select group_concat(phone order by id desc separator '_');

# strcmp(expr1, expr2)

如果expr1大于expr2，返回1，小于返回-1，等于返回0

select STRCMP(1,2)
-> -1

select STRCMP('test3','test2')
-> 1

# length(str) upper(str) lower(str)

字符串str的长度，转大小写

select LENGTH(' 123 ')
-> 5

select upper('Test')
-> TEST

select lower('Test')
-> test

# left(str, len) right(str, len)

从左边或右边开始截取指定长度的字符串

select left('test',2)
-> te

反转字符串

# row_count()

返回上一句sql增删改影响的记录数，查询返回-1

# found_rows()

上一句select或show语句的结果集的记录数

# if(expr1, expr2, expr3)

如果expr1为true则返回expr2，否则返回expr3

select if(1>2,'ture','false');-> false

# ifnull(expr1, expr2)

expr1不为null则返回expr1，为null则返回expr2

select ifnull('test1','test2')-> test1

# to_days(date)

从0开始到指定日期过了多少天

select to_days('2021-10-10')-> 738438

# datediff(date1, date2)

返回两个日期之间的天数，可以带具体时间

select datediff('2021-10-13','2021-10-25')-> -12select datediff('2021-10-13 23:59:59','2021-10-25 00:00:00')-> -12

# last_day(date)

获取该月最后一天对应的值

select last_day('2021-02-18 12:42:22')-> 2021-02-28

# date_sub/date_add(date, Interval expr1 type)

从日期里面减去指定的事件间隔，type的取值有MICROSECOND SECOND、MINUTE、HOUR、DAY、WEEK、MONTH、QUARTER、YEAR、SECOND_MICROSECOND、MINUTE_MICROSECOND、MINUTE_SECOND、HOUR_MICROSECOND、HOUR_SECOND、HOUR_MINUTE、DAY_MICROSECOND、DAY_SECOND、DAY_MIUTE、DAY_HOUR、YEAR_MONTH

select DATE_ADD('2021-10-10 02:30:23',INTERVAL 2 DAY)-> 2021-10-12 02:30:23

# format(x, d)

格式化小数点（默认进行四舍五入）

select format(23.123,2)-> 23.12

# date_format(date, format)

用于显示不同格式的日期数据，date为日期类型

select date_format(now(), '%Y-%m-%d')-> 2021-10-10

# timestampdiff(interval, datetime_expr1, datetime_expr2)

两个日期之间的整数差

select timestampdiff(hour,'2021-12-01 02:23:44','2021-12-03 00:00:00')-> 45

# least/greatest(value1, value2, ...)

返回参数中最小/最大的数或字母

# find_in_set(str, strlist)

查询str在strlist中的位置，从1开始，0表示未找到

select find_in_set('2','2,4,5,6,1');-> 1

# unix_timestamp([date])

将时间转化为时间戳，无参是表示当前时间now()

select unix_timestamp('2021-10-10 23:22:23')-> 1633879343

# from_unixtime(unix_timestamp)

格式化一个时间戳

select from_unixtime(1633879343)-> 2021-10-10 23:22:23

# coalesce(value, ...)

返回列表中第一个非NULL值

# instr(str, substr)

返回substr在字符串str第一个出现位置，没有则返回0

select instr('foobarbar','bar')-> 4

# locate(substr, str, [pos])

返回substr在str中第一次出现的位置，和instr功能相同，可自定义从哪个位置开始查找

# round(X, [D])

返回参数X最近的整数，D可选表示保留小数点后几位

# substring(str, pos, [len])

截取字符串，指定开始位置，可选长度

select substring('hello',2,3)-> ell
